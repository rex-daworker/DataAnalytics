                        $('.menu-toggle').removeClass('open');

            })

    // dropdown menu mobile
    $('.nav-group__title').on('click', function(e){
        if ($(window).width() <= 779) {
            let el = $(e.target).has('span.nav-group__triangle') ? $(e.currentTarget) : $(e.target);
            let allNavGroups = $('.nav-group');
            let navGroup = el.parents('.nav-group');

            if (navGroup.hasClass('mobile-expanded')) {

                navGroup.removeClass('mobile-expanded');
                el.find('span.nav-group__triangle').css('transform', 'rotate(0)');
                // el.siblings('.nav-group__list').css('display', 'none');
                el.siblings('.nav-group__list').removeClass('df').addClass('dn');

            } else if (!navGroup.hasClass('mobile-expanded')) {

                allNavGroups.removeClass('mobile-expanded');
                $('.nav-group__title span.nav-group__triangle').css('transform', 'rotate(0)');
                // $('.nav-group__list').css('display', 'none');
                $('.nav-group__list').removeClass('df').addClass('dn');

                navGroup.addClass('mobile-expanded');
                el.find('span.nav-group__triangle').css('transform', 'rotate(180deg)');
                // el.siblings('.nav-group__list').css('display', 'flex');
                el.siblings('.nav-group__list').removeClass('dn').addClass('df');
            }
        }
    });

    // resetting mobile collapsible menu after resizing
    $(window).on('resize', function () {
            $('.nav-group').removeClass('mobile-expanded');
            $('.nav-group__list').removeClass('df dn');
            $('.nav-group__title span.nav-group__triangle').css('transform', 'rotate(0)');
    });
