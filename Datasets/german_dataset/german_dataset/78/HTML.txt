<script>
    $(function(){

        let gameBlockButtons = $('.game-block__buttons'),
             gameBlockResize = $('.game-block__resize'),
             iframeGame = $('.iframe-game-new'),
             gameBlockContainer = $('.game-block__container'),
             overlayWrap = $('.game-overlay'),
             newClose = $('.game-block__close'),
             gameBlockIframePlay = $('.game-block__iframe-play');

        let gameBlockRatio = 1.405,
             fullscreenWidth = 1200,
            mobileWidth = 700;

        let isFullscreen = 0;
        let isLoaded = false;

        let vh = window.innerHeight * 0.01;

        document.documentElement.style.setProperty('--vh', `${vh}px`);

        (function addDataSrcToIframe() {

            $('.game-launcher__btn-play-free').on('click', function (ev) {

                

                overlayWrap.addClass("active");

                if ($(window).width() <= 800) {
                    newClose.click()
                }

                showResizeButton();
                hideGameScreenButtons();

                if (typeof page !== 'undefined' && page !== 'roulette-simulator-page'){
                    setBorderRadius(20);
                }

                if (typeof page !== 'undefined'  && page !== 'roulette-simulator-page'){
                    hideBackgrounds();
                } else {
                    hideBackgrounds_simulator();
                }

                let src = window.innerWidth < 700 ? ev.target.getAttribute('data-mobile-src') : ev.target.getAttribute('data-src');
                $('.iframe-game, .iframe-game-mobile').attr('src', src);
            });
        })();

        function hideGameScreenButtons() { gameBlockButtons.hide(); }
        function showGameScreenButtons() { gameBlockButtons.show(); }
        function showResizeButton(){ gameBlockResize.show(); }
        function hideResizeButton(){gameBlockResize.hide();}

        function setBorderRadius(value){
            iframeGame.css('borderRadius', value);
            gameBlockContainer.css('borderRadius', value);
        }

        function hideBackgrounds(){
                gameBlockContainer.css('backgroundSize', 0);
                iframeGame.css('background', 'none');
        }

        function showBackgrounds(){
            gameBlockContainer.css('backgroundSize', 'cover');
            iframeGame.css('background', 'rgba(0,0,0,0.4)');
        }

        function updateResizeBtnPosition(isUnset){
            if (isUnset === 'unset'){
                gameBlockResize.css('right', 0);
            } else {
                let windowWidth = $(window).width(),
                     iframeWidth = $('.iframe-game-new').width(),
                     offsetWidth = (windowWidth - iframeWidth) / 2 + 100;
                $('.game-block__resize.fullscreen').css('right', offsetWidth);
            }
        }

        function updateFullscreenPlayPosition(isUnset){
            if (isUnset === 'unset'){
                gameBlockIframePlay.css('display', 'none');
                return;
            }
            let fullscreenMaxHeight = 736;      
            let topPosition = 100 * vh - 40 + 'px';
            gameBlockIframePlay.css({
                'display': 'block',
                'top': topPosition
            });
        }

        overlayWrap.on('click', function() {
            if (!isFullscreen) {
                setFullScreen();
                updateResizeBtnPosition();
                updateFullscreenPlayPosition();
                isFullscreen = 1;
            } else if (isFullscreen) {
                exitFullScreen();
                updateResizeBtnPosition('unset');
                updateFullscreenPlayPosition('unset');
                isFullscreen = 0;
            }
        })

        function setFullScreen(){
            iframeGame.addClass('fullscreen');
            gameBlockResize.addClass('fullscreen');
            overlayWrap.css('display', 'block');
            $('body').css('overflow', 'hidden');
            if ($(window).width() <= 800) {
                newClose.css('display', 'flex');
                $('.game-block-new').addClass('horizont');
                hideResizeButton()
            }
        }

        function exitFullScreen(){
            iframeGame.removeClass('fullscreen');
            gameBlockResize.removeClass('fullscreen');
            overlayWrap.css('display', 'none');
            $('body').css('overflow', 'auto');
            if ($(window).width() <= 800) {
                newClose.css('display', 'none');
                showResizeButton();
                $('.game-block-new').removeClass('horizont');
            }
        }

        newClose.on('click', function() {
            exitFullScreen();
            updateResizeBtnPosition('unset');
            updateFullscreenPlayPosition('unset');
            isFullscreen = 0;
            $('.game-block-new').removeClass('horizont');
            overlayWrap.removeClass('active');
        })

        function loadIframe() {
            if (!isLoaded) {
                var srcIframe = $('.block-video__iframe').attr('data-src');
                $(".block-video__iframe").attr('src', srcIframe);
                isLoaded = true;
            }
        }
        if ($(window).height() > 1045) {loadIframe();}

        $(window).scroll(function () {loadIframe();});

        //handles the game choosing at game pages
        (function chooseGameViaTabulation() {


            function handleTabulation() {

                // click on tab (simulator page)
                $('.game-block__tab-item').on('click', function(){


                    if (!$(this).hasClass('active')){
                        $('.game-block__tab-item').removeClass('active');
                        $(this).addClass('active');
                        setInactiveTabWidth();
                    }



                    showGameScreenButtons();
                    // hideResizeButton();
                    showBackgrounds();

                    $('.iframe-game-new').attr('src', '');

                    let newTabButtons = $(this).find('button'),
                         newTab;

                    newTabButtons.each(function(i,btn){
                            if ($(window).innerWidth() > mobileWidth && i === 0) {
                                newTab = $(btn);
                                newTab.addClass('is-active');
                            } else if ($(window).innerWidth() <= mobileWidth && i === 1) {
                                newTab = $(btn);
                                newTab.addClass('is-active');
                            } else $(btn).removeClass('is-active');
                    });

                    console.log(newTab);

                    let preloadImage = newTab.attr('data-game-preload-img');

                    $('.game-launcher__btn-play-free').attr('data-src', newTab.attr('data-src')); //take attributes from tab and place to the free game btn

                    if (newTab.attr('data-mobile-src')) {
                        $('.game-launcher__btn-play-free').attr('data-mobile-src', newTab.attr('data-mobile-src'));
                    } else {
                        $('.game-launcher__btn-play-free').attr('data-mobile-src', '');
                    }

                    $('.game-block__container').css('background-image', 'url(' + preloadImage + ')');
                    $('.iframe-game-new').css('background','');
                    return false;
                });



                /// click on button (eu am pages)
                $('.game-tabs__btn').on('click', function () {

                    $(this).siblings().removeClass('is-active').end().addClass('is-active'); //mark active\non-active tab state

                    showGameScreenButtons();
                    hideResizeButton();
                    showBackgrounds();

                    $('.iframe-game-new').attr('src', '');

                    $('.game-launcher__btn-play-free').attr('data-src', $(this).attr('data-src')); //take attributes from tab and place to the free game btn
                    $('.game-launcher__btn-play-free').attr('data-mobile-src', $(this).attr('data-mobile-src'));
                    $('.game-block__container').css('background-image', 'url(' + $(this).attr('data-game-preload-img') + ')');
                    $('.iframe-game-new').css('background','');
                    return false;

                })
            }
            handleTabulation();

            function handleMobGame() {
                if ($(window).innerWidth() < mobileWidth) {
                    let btn = document.querySelectorAll('.game-tabs__btn');
                    btn.forEach(item => {
                        item.classList.remove('is-active');
                    })
                    $('#priorityMobGame').addClass('is-active');
                    //приоритетная кнопка
                    if ( window.innerWidth < 700 && $('#priorityMobGame').hasClass('is-active')) {
                        $('.game-launcher__btn-play-free').attr('data-mobile-src', $('#priorityMobGame').attr('data-mobile-src'));
                        $('.game-block__container').css('background-image', 'url(' + $('#priorityMobGame').attr('data-game-preload-img') + ')');
                    }
                }
            }
            handleMobGame()
        })();


        gameBlockResize.on('click', function(){
            if (!isFullscreen) {
                setFullScreen();
                updateResizeBtnPosition();
                updateFullscreenPlayPosition();
                isFullscreen = 1;
            } else if (isFullscreen) {
                exitFullScreen();
                updateResizeBtnPosition('unset');
                updateFullscreenPlayPosition('unset');
                isFullscreen = 0;
            }
        });


        $(window).on('resize', function () {

            if (window.orientation == 0 || window.orientation == 180) {
                vh = window.innerHeight * 0.01;
            } else {
                vh = window.innerHeight * 0.01;
            }

            document.documentElement.style.setProperty('--vh', `${vh}px`);

            gameBlockContainer.height(gameBlockContainer.width() / gameBlockRatio);

            if (isFullscreen){
                updateResizeBtnPosition();
                updateFullscreenPlayPosition();
            }


        });
        $(window).trigger('resize');

        $(document).on('keyup', function(e){
            if (e.keyCode === 27 && isFullscreen) {
                gameBlockResize.click();
            }
        });


        function setInactiveTabWidth(){
            let iframeW = $('.iframe-game').width(),
                activeTabW = $('.game-block__tab-item.active').width();

            $('.game-block__tab-item').not('.active').css('width', '100%');
            $('.game-block__tab-item.active').css('width', 'auto');
        }
        setInactiveTabWidth();



        function hideBackgrounds_simulator(){
            gameBlockContainer.css('background', '#054F47');
            iframeGame.css('background', 'none');
        }

    });
</script>

<script>
    jQuery(document).ready( function ($) {

        $(".accordeon__row:nth-child(1) .accordeon__title").addClass('active');
        $(".accordeon__row:nth-child(1) .accordeon__text").css('display', 'block');
        $('.accordeon .accordeon__title').on('click', f_acc);

        function f_acc() {

            $(".accordeon__row:nth-child(1) .accordeon__title").removeClass('active');
            $(".accordeon__row:nth-child(1) .accordeon__text").css('display', 'none');
            $('.accordeon .accordeon__text').not($(this).next()).slideUp(400);
            $(this).next().slideToggle(500);
            $(this).toggleClass('active');
            $('.accordeon__title').not($(this)).removeClass('active');
            $(this).parent().next().slideDown();
            return false;
        }
    });
</script>
